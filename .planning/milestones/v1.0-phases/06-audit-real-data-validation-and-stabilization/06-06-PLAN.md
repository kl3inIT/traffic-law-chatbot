// frontend/e2e/chat.spec.ts
import { test, expect } from '@playwright/test';

// Helper: send a single-turn question and wait for AI response
async function sendSingleTurnQuestion(page, question: string) {
  await page.goto('/');
  const textarea = page.getByPlaceholder('Nhập câu hỏi về luật giao thông...');
  await textarea.fill(question);
  await textarea.press('Enter');
  // Wait for AI bubble to appear (AiBubbleLoading disappears, real content renders)
  await expect(page.locator('[data-from="assistant"]').last()).not.toContainText('', { timeout: 30000 });
  // Wait for skeleton to disappear
  await page.waitForSelector('.animate-pulse', { state: 'detached', timeout: 30000 }).catch(() => {});
  return page.locator('[data-from="assistant"]').last();
}

// ─── GROUNDED RESPONSE TESTS ──────────────────────────────────────────────────

test('T-UI-01: grounded answer renders disclaimer text', async ({ page }) => {
  const bubble = await sendSingleTurnQuestion(page, 'Xe máy vượt đèn đỏ bị phạt bao nhiêu?');
  // Disclaimer is in MessageToolbar — verify text presence
  await expect(bubble).toContainText('tham khảo', { timeout: 10000 });
});

test('T-UI-02: grounded answer renders citation list', async ({ page }) => {
  const bubble = await sendSingleTurnQuestion(page, 'Không đội mũ bảo hiểm xe máy bị phạt bao nhiêu?');
  // Citations render as [Nguồn N] labels or source titles
  await expect(bubble).toContainText('Nguồn', { timeout: 10000 });
});

test('T-UI-03: grounded answer renders legal basis section', async ({ page }) => {
  const bubble = await sendSingleTurnQuestion(page, 'Uống rượu lái xe máy bị phạt thế nào theo Nghị định 168?');
  // "Căn cứ pháp lý" section header
  await expect(bubble).toContainText('Căn cứ pháp lý', { timeout: 10000 });
});

test('T-UI-04: grounded answer does not fabricate content — disclaimer always present', async ({ page }) => {
  const bubble = await sendSingleTurnQuestion(page, 'Cần mang theo giấy tờ gì khi tham gia giao thông?');
  // Both legal basis and disclaimer must be present for grounded response
  await expect(bubble).toContainText('Căn cứ pháp lý', { timeout: 10000 });
  await expect(bubble).toContainText('tham khảo', { timeout: 10000 });
});

// ─── CLARIFICATION NEEDED TESTS ───────────────────────────────────────────────

test('T-UI-05: clarification needed — amber box renders when vehicleType missing', async ({ page }) => {
  const bubble = await sendSingleTurnQuestion(page, 'Vượt đèn đỏ bị phạt bao nhiêu?');
  // Amber "Cần làm rõ thêm" box appears for CLARIFICATION_NEEDED
  await expect(bubble).toContainText('Cần làm rõ thêm', { timeout: 10000 });
});

test('T-UI-06: clarification needed — pending fact prompt is visible', async ({ page }) => {
  const bubble = await sendSingleTurnQuestion(page, 'Bị phạt bao nhiêu nếu không có đăng ký xe?');
  // Pending fact prompt text about vehicle type
  await expect(bubble).toContainText('phương tiện', { timeout: 10000 });
});

// ─── REFUSED RESPONSE TESTS ───────────────────────────────────────────────────

test('T-UI-07: refused response — off-topic question renders refusal without legal content', async ({ page }) => {
  const bubble = await sendSingleTurnQuestion(page, 'Luật hôn nhân gia đình quy định gì?');
  // Should NOT contain legal basis section (would be hallucination)
  await expect(bubble).not.toContainText('Căn cứ pháp lý', { timeout: 10000 });
  // Should NOT render a citation list (no legal sources cited)
  await expect(bubble).not.toContainText('[Nguồn 1]');
});

test('T-UI-08: hallucination probe — non-existent article returns refusal, not invented content', async ({ page }) => {
  const bubble = await sendSingleTurnQuestion(page, 'Theo Điều 99b Nghị định 168/2024, mức phạt là bao nhiêu?');
  // Must NOT contain "99b" as a cited article — that would be hallucination
  await expect(bubble).not.toContainText('Điều 99b');
});

// ─── MULTI-TURN THREAD TESTS ──────────────────────────────────────────────────

test('T-UI-09: multi-turn thread — new thread appears in thread list', async ({ page }) => {
  await page.goto('/');
  // Click "New chat" or equivalent to create a new thread
  const newChatBtn = page.getByRole('button', { name: /new|mới|tạo/i }).first();
  if (await newChatBtn.isVisible()) {
    await newChatBtn.click();
  }
  const textarea = page.getByPlaceholder('Nhập câu hỏi về luật giao thông...');
  await textarea.fill('Xe máy không đội mũ bảo hiểm bị phạt bao nhiêu?');
  await textarea.press('Enter');
  await page.waitForSelector('.animate-pulse', { state: 'detached', timeout: 30000 }).catch(() => {});
  // Thread list should update — check for at least one thread item
  await expect(page.locator('nav, aside').first()).toBeVisible();
});

test('T-UI-10: multi-turn thread — second message receives clarification response building on first', async ({ page }) => {
  await page.goto('/');
  const textarea = page.getByPlaceholder('Nhập câu hỏi về luật giao thông...');

  // First message — starts thread
  await textarea.fill('Tôi đang bị CSGT dừng xe');
  await textarea.press('Enter');
  await page.waitForSelector('.animate-pulse', { state: 'detached', timeout: 30000 }).catch(() => {});

  // First response should ask for clarification
  const firstBubble = page.locator('[data-from="assistant"]').first();
  await expect(firstBubble).toBeVisible({ timeout: 30000 });

  // Second message — provides vehicle type
  await page.getByPlaceholder('Nhập câu hỏi về luật giao thông...').fill('Tôi đi xe máy');
  await page.getByPlaceholder('Nhập câu hỏi về luật giao thông...').press('Enter');
  await page.waitForSelector('.animate-pulse', { state: 'detached', timeout: 30000 }).catch(() => {});

  // Second assistant bubble should exist (thread continued)
  const bubbles = page.locator('[data-from="assistant"]');
  await expect(bubbles).toHaveCount(2, { timeout: 30000 });
});

INTERPRET RESULTS:

For each failing test, determine the root cause category:

A. **Selector mismatch** — the test assumed an element attribute (e.g., `data-from="assistant"`) that does not exist in the actual DOM. Fix by reading the actual HTML and updating the selector. Run `npx playwright test {test-name} --headed --reporter=list` to see the rendered page.

B. **Backend/frontend not connected** — test fails with network error or empty response. Verify NEXT_PUBLIC_API_URL environment variable points to http://localhost:8080. Check `frontend/.env.local`.

C. **Real behavior mismatch** — the UI renders differently than expected for a valid response (e.g., disclaimer text uses different wording than "tham khảo"). This is a BLOCKING finding — the UI is not rendering what was specified. Fix the test to match actual wording OR fix the component if the wording is wrong.

D. **Hallucination detected** — T-UI-08 fails because "Điều 99b" appears in the response. This is a BLOCKING finding requiring investigation via AUDIT.md (root cause: retrieval, prompt, or model behavior).

CLASSIFY ALL FAILURES:
- **BLOCKING**: any test failure where the UI renders incorrect content (hallucination, missing disclaimer on grounded response, citation appears on refused response)
- **MAJOR**: test failure due to selector mismatch where the underlying feature works but test cannot verify it
- **MINOR**: admin page loads but specific element text differs from expected (e.g., different badge label wording)

Fix MAJOR failures by correcting selectors. Fix BLOCKING failures by investigating root cause (read ChatLog.pipelineLog for the failing query). Do NOT mark a test as passing by making it trivially non-assertive.
