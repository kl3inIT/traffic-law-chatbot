```java
package com.vn.traffic.chatbot.chat.regression;

import com.vn.traffic.chatbot.chat.service.ChatService;
import com.vn.traffic.chatbot.common.config.AiModelProperties;
import org.junit.jupiter.api.Tag;
import org.junit.jupiter.api.Test;
import org.junit.jupiter.api.condition.DisabledIfEnvironmentVariable;
import org.springframework.ai.evaluation.BasicEvaluationTest;
import org.springframework.ai.evaluation.RelevancyEvaluator;
import org.springframework.ai.evaluation.FactCheckingEvaluator;
import org.springframework.ai.evaluation.EvaluationRequest;
import org.springframework.ai.evaluation.EvaluationResponse;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.boot.test.context.SpringBootTest;
// ... list imports for YAML reading + ChatClient builder ...

@SpringBootTest
@Tag("live")
@DisabledIfEnvironmentVariable(named = "OPENROUTER_API_KEY", matches = "^$")
class VietnameseRegressionIT extends BasicEvaluationTest {

    @Autowired ChatService chatService;
    @Autowired AiModelProperties props;
    @Autowired java.util.Map<String, org.springframework.ai.chat.client.ChatClient> chatClientMap;

    private static final int MIN_PASS_PERCENT = 95;
    private static final int P7_REFUSAL_BASELINE_PERCENT = 20;   // UPDATE: read from 07-SUMMARY.md
    private static final int REFUSAL_DRIFT_TOLERANCE_PERCENT = 10;

    @Test
    void twentyQueryRegressionSuiteAtLeast95Percent() {
        var queries = loadFixture();   // parse vietnamese-queries-20.yaml
        var evaluator = new RelevancyEvaluator(chatClientMap.get(props.chatModel()));
        var factChecker = new FactCheckingEvaluator(chatClientMap.get(props.chatModel()));
        int passed = 0;
        for (var q : queries) {
            var resp = chatService.answer(q.question(), props.chatModel(), /*conversationId*/ null);
            String citationsText = String.join("\n", resp.citations().stream().map(c -> c.snippet()).toList());
            var relevancy = evaluator.evaluate(new EvaluationRequest(q.question(), java.util.List.of(citationsText), resp.answer()));
            var grounding = factChecker.evaluate(new EvaluationRequest(citationsText, java.util.List.of(), resp.answer()));
            if (relevancy.isPass() && grounding.isPass()) passed++;
        }
        double percent = passed * 100.0 / queries.size();
        org.assertj.core.api.Assertions.assertThat(percent)
            .as("Regression pass rate")
            .isGreaterThanOrEqualTo(MIN_PASS_PERCENT);
    }

    @Test
    void refusalRateWithinTenPercentOfPhase7Baseline() {
        var queries = loadFixture();
        int refused = 0;
        for (var q : queries) {
            var resp = chatService.answer(q.question(), props.chatModel(), null);
            if (resp.groundingStatus() == com.vn.traffic.chatbot.chat.service.GroundingStatus.REFUSED) refused++;
        }
        double rate = refused * 100.0 / queries.size();
        org.assertj.core.api.Assertions.assertThat(Math.abs(rate - P7_REFUSAL_BASELINE_PERCENT))
            .as("Refusal rate %.2f%% must be within ±%d%% of P7 baseline %d%%",
                rate, REFUSAL_DRIFT_TOLERANCE_PERCENT, P7_REFUSAL_BASELINE_PERCENT)
            .isLessThanOrEqualTo(REFUSAL_DRIFT_TOLERANCE_PERCENT);
    }

    @Test
    void twoTurnConversationMemoryWorks() {
        String convId = java.util.UUID.randomUUID().toString();
        var r1 = chatService.answer("Vượt đèn đỏ phạt bao nhiêu?", props.chatModel(), convId);
        var r2 = chatService.answer("Còn xe máy thì sao?", props.chatModel(), convId);
        org.assertj.core.api.Assertions.assertThat(r2.answer()).isNotBlank();
        // Turn 2 should not 400 (role-alternation Pitfall 3) and should reference turn-1's red-light context.
        // Soft signal: answer mentions "đèn đỏ" or "vượt" (turn-1 topic).
        org.assertj.core.api.Assertions.assertThat(r2.answer().toLowerCase())
            .containsAnyOf("đèn đỏ", "vượt", "xe máy");
    }

    private java.util.List<QueryFixture> loadFixture() { /* parse YAML */ }
    record QueryFixture(String id, String question, String expectedCitationHint) {}
}
```
