export function useCreateThread() {
  const queryClient = useQueryClient();
  return useMutation({
    mutationFn: ({ question, modelId }: { question: string; modelId?: string }) =>
      createThread(question, modelId),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: queryKeys.threads });
    },
  });

**Step 7: Convert chat layout to client component + add model context (per D-16)**

Replace `frontend/app/(chat)/layout.tsx` with:
```tsx
'use client';

import { createContext, useContext, useState } from 'react';
import { useAllowedModels } from '@/hooks/use-parameters';
import type { AllowedModel } from '@/types/api';

interface ChatModelContextValue {
  selectedModelId: string | undefined;
  setSelectedModelId: (id: string) => void;
  allowedModels: AllowedModel[];
  isLoadingModels: boolean;
}

const ChatModelContext = createContext<ChatModelContextValue>({
  selectedModelId: undefined,
  setSelectedModelId: () => {},
  allowedModels: [],
  isLoadingModels: false,
});

export function useChatModel() {
  return useContext(ChatModelContext);
}

export default function ChatLayout({ children }: { children: React.ReactNode }) {
  const { data: allowedModels = [], isLoading: isLoadingModels } = useAllowedModels();
  const [selectedModelId, setSelectedModelId] = useState<string | undefined>(undefined);

  return (
    <ChatModelContext.Provider
      value={{ selectedModelId, setSelectedModelId, allowedModels, isLoadingModels }}
    >
      <div className="flex h-screen">{children}</div>
    </ChatModelContext.Provider>
  );
}
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Browser → GET /api/v1/admin/allowed-models | Frontend fetches model list from backend |
| User → chat request with modelId | User can choose any model from the dropdown |
| Admin → parameter set update with chatModel/evaluatorModel | Admin sets evaluator model for check runs |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-06.1-06 | Tampering | Frontend chat request — modelId field | accept | modelId is populated from the allowed-models endpoint. A user who manually crafts a request with an arbitrary modelId will hit the backend D-12 warn+fallback behavior. No client-side validation needed. |
| T-06.1-07 | Information Disclosure | Model list visible in browser network tab | accept | Model IDs and display names are non-sensitive operational information. No credentials or internal configuration exposed. |
| T-06.1-08 | Spoofing | Context state lost on browser refresh | accept | selectedModelId resets to undefined on refresh — treated as "use default". This is intentional and harmless; the fallback model is always valid. |
</threat_model>
